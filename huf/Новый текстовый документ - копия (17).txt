string GetTextDiv(string text)
{
    string output = "&nbsp";
    string[] copy = text.Split(' ');

    if (copy.Length == 1)
    {
        return output + copy[0];
    }

    for (int i = 0; i < copy.Length / 2 + 1; i++)
    {
        output += copy[i] + " ";
    }

    output += "<br/>&nbsp";

    for (int i = copy.Length / 2 + 1; i < copy.Length; i++)
    {
        output += copy[i] + " ";
    }
    return output;
}
